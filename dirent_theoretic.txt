struct dirent
{
	ino_t          d_ino;       // Номер inode файла (уникальный идентификатор файла в файловой системе)
    off_t          d_off;       // Смещение к следующей записи в директории (не всегда используется)
    unsigned short d_reclen;    // Длина записи
    unsigned char  d_type;      // Тип файла (см. ниже)
    char           d_name[256]; // Имя файла (строка, завершается нулём)
}
